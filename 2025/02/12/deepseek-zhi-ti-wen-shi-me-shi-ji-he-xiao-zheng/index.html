例如，假设原图的某个点的坐标为 (src_x, src_y)，在数组中对应的位置是 [y, x]。例如，src_x是列数，对应到数组的第二个轴，src_y是行数，对应到数组的第一个轴。这可能容易出错。

例如，如果原图是宽度为W，高度为H，那么数组的索引是 [y][x]，其中x的范围是0到W-1，y的范围是0到H-1。所以，当计算得到原图的坐标为 (src_x, src_y)，需要转换为数组的索引为 (int(src_y), int(src_x))。或者，可能坐标的x对应数组的列（第二个轴），y对应数组的行（第一个轴）。

这可能是一个容易出错的地方。例如，在代码中，当用户提供的控制点中的src_x和src_y是原图中的坐标，比如原图的左上角是 (0,0)，向右为x增加，向下为y增加。那么在数组中，对应的是 [y][x]。所以在处理时，比如，原图的坐标是 (x, y)，对应的数组索引是 [y][x]。例如，src_x的范围是 0到width_src-1，src_y的范围是0到height_src-1。

所以，在双线性插值时，原图的坐标需要转换为数组的索引，即：

x0 = floor(src_x)

y0 = floor(src_y)

x1 = x0 + 1

y1 = y0 + 1

然后，检查这些x0, x1是否在 [0, width_src -1] 范围内，y0, y1是否在 [0, height_src -1] 范围内。如果越界，则处理为0或者跳过。

但是，在代码中，原图的宽度和高度是从src_img.size得到的，而src_img.size是 (width, height)。所以在数组中的形状是 (height, width, channels)。因此，在访问数组时，必须确保y0和y1不超过height-1，x0和x1不超过width-1。

#!/usr/bin/env python
# -*- coding: utf-8 -*- 
# @Time : 2025/2/12 

'''

'''
import numpy as np
from PIL import Image
import math

def bilinear_interpolation(src_array, x, y):
    """双线性插值实现"""
    x0, y0 = math.floor(x), math.floor(y)
    x1, y1 = x0 + 1, y0 + 1

    # 边界检查
    height, width = src_array.shape[:2]
    if not (0 <= x0 < width and 0 <= x1 < width and
            0 <= y0 < height and 0 <= y1 < height):
        return 0

    # 计算权重
    dx, dy = x - x0, y - y0

    # 获取四个邻域像素值
    if len(src_array.shape) == 3:  # 彩色图像
        top_left = src_array[y0, x0, :]
        top_right = src_array[y0, x1, :]
        bottom_left = src_array[y1, x0, :]
        bottom_right = src_array[y1, x1, :]
    else:  # 灰度图像
        top_left = src_array[y0, x0]
        top_right = src_array[y0, x1]
        bottom_left = src_array[y1, x0]
        bottom_right = src_array[y1, x1]

    # 插值计算
    top = top_left * (1 - dx) + top_right * dx
    bottom = bottom_left * (1 - dx) + bottom_right * dx
    return top * (1 - dy) + bottom * dy


def polynomial_coefficients(control_points, order):
    """计算多项式变换系数"""
    dst_points = np.array([[pt[0], pt[1]] for pt in control_points])
    src_x = np.array([pt[2] for pt in control_points])
    src_y = np.array([pt[3] for pt in control_points])

    # 构建多项式矩阵
    A = []
    for (x, y) in dst_points:
        terms = [1]
        if order >= 1:  # 一阶项
            terms += [x, y]
        if order >= 2:  # 二阶项
            terms += [x ** 2, x * y, y ** 2]
        A.append(terms)

    A = np.array(A)
    coeffs_x = np.linalg.lstsq(A, src_x, rcond=None)[0]
    coeffs_y = np.linalg.lstsq(A, src_y, rcond=None)[0]
    return coeffs_x, coeffs_y


def geometric_correction(src_path, control_points, order=1, output_size=None):
    """
    几何校正主函数
    :param src_path: 原始图像路径
    :param control_points: 控制点列表，格式[(dst_x, dst_y, src_x, src_y)]
    :param order: 多项式阶数（1或2）
    :param output_size: 输出图像尺寸（宽，高）
    """
    # 读取原始影像
    src_img = Image.open(src_path)
    src_array = np.array(src_img)
    src_height, src_width = src_array.shape[:2]

    # 计算变换系数
    coeffs_x, coeffs_y = polynomial_coefficients(control_points, order)

    # 确定输出尺寸
    if output_size is None:
        output_width, output_height = src_img.size
    else:
        output_width, output_height = output_size

    # 创建输出图像
    if len(src_array.shape) == 3:
        corrected = np.zeros((output_height, output_width, 3), dtype=np.uint8)
    else:
        corrected = np.zeros((output_height, output_width), dtype=np.uint8)

    # 遍历输出图像的每个像素
    for y in range(output_height):
        for x in range(output_width):
            # 计算对应的原图坐标
            terms = [1]
            if order >= 1:
                terms += [x, y]
            if order >= 2:
                terms += [x ** 2, x * y, y ** 2]

            src_x = np.dot(terms, coeffs_x)
            src_y = np.dot(terms, coeffs_y)

            # 执行插值
            src_array = src_array[:,:,:3]
            if 0 <= src_x < src_width and 0 <= src_y < src_height:
                corrected[y, x] = bilinear_interpolation(src_array, src_x, src_y)

    return Image.fromarray(corrected)


# 示例使用
if __name__ == "__main__":
    # 定义控制点示例（目标坐标X，目标坐标Y，原图坐标X，原图坐标Y）
    control_points = [
        (0, 0, 0, 50),  # 左上角
        (300, 0, 250, 30),  # 右上角
        (0, 300, 40, 250),  # 左下角
        (600, 500, 560, 570)  # 右下角
    ]

    # 执行几何校正
    corrected_img = geometric_correction(
        "11.png",
        control_points,
        order=1,
        output_size=(1500, 1500)
    )
    corrected_img.save("corrected.png")
