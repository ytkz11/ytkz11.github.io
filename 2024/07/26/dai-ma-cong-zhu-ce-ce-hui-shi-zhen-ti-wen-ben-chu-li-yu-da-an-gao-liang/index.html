def parse_text(text):
    '''
    з”ЁдәҺ2011е№ҙеҲ°2021е№ҙжөӢз»ҳзңҹйўҳ
    :param text: ж–Үжң¬
    :return: йўҳзӣ®йҖүйЎ№зӯ”жЎҲзҡ„еҲ—иЎЁ
    '''
    text = text.replace('гҖҒ', '.')
    # жӯЈеҲҷиЎЁиҫҫејҸпјҢз”ЁдәҺеҲҶеүІжҜҸдёӘй—®йўҳзҡ„ж®өиҗҪ
    problem_pattern = re.compile(r'(\d+\..*?(?=^(\d+\.|$)))', re.MULTILINE | re.DOTALL)

    # жӯЈеҲҷиЎЁиҫҫејҸпјҢз”ЁдәҺеңЁжҜҸдёӘй—®йўҳзҡ„ж®өиҗҪдёӯжҸҗеҸ–йўҳзӣ®гҖҒйҖүйЎ№гҖҒзӯ”жЎҲе’Ңи§Јжһҗ
    all_pattern =            re.compile(r'^(.*?)\s*(A[.,\гҖҒ]*.*?E[.,\гҖҒ]*.*?)(зӯ”жЎҲпјҡ[\w, ]+)(и§Јжһҗпјҡ.*)', re.DOTALL)
    no_explanation_pattern = re.compile(r'^(.*?)\s*(A[.,\гҖҒ]*.*?D[.,\гҖҒ]*.*?)(зӯ”жЎҲпјҡ[\w, ]+)', re.DOTALL)

    problems = problem_pattern.findall(text)
    parsed_problems = []
    i = 0
    for problem in problems:
        problem = list(problem)[0]
        match_all = all_pattern.match(problem)
        match_no_explanation = no_explanation_pattern.match(problem)

        match = match_all if match_all else match_no_explanation
        i+=1
        if i == 30:
            print()
        if match:
            question = match.group(1).strip()

            #  дҝ®ж”№еҗҺзҡ„жӯЈеҲҷиЎЁиҫҫејҸпјҢзЎ®дҝқйҖүйЎ№еүҚжңүз©әж јжҲ–\nпјҢе№¶дё”зҙ§и·ҹдёҖдёӘеӨ§еҶҷеӯ—жҜҚе’ҢвҖңгҖҒвҖқ
            origin_option = match.group(2).replace('\n', ' ').strip()
            options_pattern = re.compile(r'(?<=\s|\n)[A-F][.,\гҖҒ]*\s*[^зӯ”жЎҲ]+')
            options_pattern = re.compile(r'(?<=\s|\n)[A-F][.,\гҖҒ]\s*\S.*?(?= [A-F]|$)')
            options_pattern = re.compile(r'[A-F][.,\гҖҒ]*\s*\S.*?(?=\S[A-F]|$)')

            options_matches = options_pattern.findall(origin_option)

            options = [option.strip() for option_match in options_matches for option in option_match.split()]

            answer = match.group(3).replace('зӯ”жЎҲпјҡ', '').strip().replace(' ', '')
            explanation = match.group(4).strip() if match_all else ''

            parsed_problems.append({
                'question': question.replace('\n',''),
                'options': options,
                'answer': answer,
                'explanation': explanation
            })

    return parsed_problems

def extract_options(options_str):
    # еҒҮи®ҫoptions_strзҡ„ж јејҸдёә'A.N,B.N-1,C.N(N+1)/2,D.N(N-1)/2,E.е…¶д»–еҶ…е®№'
    # еҰӮжһңжІЎжңүEпјҢеҲҷеҸӘеӨ„зҗҶеҲ°D
    options_list = options_str.rstrip().split(' ')

    # еҲқе§ӢеҢ–дёҖдёӘеӯ—е…ёжқҘеӯҳеӮЁйҖүйЎ№
    options_dict = {}

    # йҒҚеҺҶйҖүйЎ№еҲ—иЎЁ
    for option in options_list:
        # еҲҶеүІйҖүйЎ№дёәеӯ—жҜҚе’ҢеҶ…е®№
        letter, content = option.split('.', 1)
        options_dict[letter.strip()] = content.strip()

        # еҰӮжһңжІЎжңүEпјҢеҲҷеҸӘеҸ–A, B, C, D
    # еҰӮжһңжңүEпјҢеҲҷзӣҙжҺҘиҝ”еӣһжүҖжңүйҖүйЎ№
    if 'E' in options_dict:
        # еҰӮжһңйңҖиҰҒпјҢеҸҜд»ҘеңЁиҝҷйҮҢеҜ№options_dictиҝӣиЎҢиҝӣдёҖжӯҘеӨ„зҗҶжҲ–иҝ”еӣһ
        return options_dict
    else:
        # жҸҗеҸ–A, B, C, D
        return {k: v for k, v in options_dict.items() if k in ['A', 'B', 'C', 'D']}

def split_text(text):

    # жӯЈеҲҷиЎЁиҫҫејҸз”ЁдәҺеҢ№й…Қйўҳзӣ®гҖҒйҖүйЎ№гҖҒзӯ”жЎҲе’Ңи§Јжһҗ
    pattern = r'(?P<question>\d+\..*?\n(?:A|B|C|D).*\n)зӯ”жЎҲпјҡ(?P<answer>[A-D])\nи§Јжһҗпјҡ(?P<explanation>.*?(?=\n\d+\.|$))'
    questions = []

    # дҪҝз”Ёre.finditerжқҘжҹҘжүҫжүҖжңүеҢ№й…ҚйЎ№
    for match in re.finditer(pattern, text, re.MULTILINE | re.DOTALL):
        question = match.group('question').strip()
        options = re.findall(r'(A|B|C|D)\..*', question)  # жҸҗеҸ–йҖүйЎ№пјҲз®ҖеҢ–еӨ„зҗҶпјҢеҸҜиғҪйңҖиҰҒж №жҚ®е®һйҷ…жғ…еҶөи°ғж•ҙпјү
        answer = match.group('answer')
        explanation = match.group('explanation').strip()

        # е°Ҷйўҳзӣ®гҖҒйҖүйЎ№гҖҒзӯ”жЎҲе’Ңи§Јжһҗж·»еҠ еҲ°еҲ—иЎЁдёӯпјҲиҝҷйҮҢдёәдәҶз®ҖеҚ•иө·и§ҒпјҢеҸӘж·»еҠ дәҶйўҳзӣ®е’Ңзӯ”жЎҲпјҢдҪҶеҸҜд»Ҙжү©еұ•дёәеҢ…еҗ«йҖүйЎ№е’Ңи§Јжһҗзҡ„еӯ—е…ёпјү
        questions.append({
            'question': question,
            'options': options,
            'answer': answer,
            'explanation': explanation
        })
    return questions
