粗提纯

找到SIFT特征，用比率检测来找最匹配的
原始的 SIFT 算法中，匹配提纯方法使用的相似性测度是欧氏距离
使用最近/次近邻欧氏距离比实现匹配点对的提纯
用三个最接近的特征点进行距离匹配

计算rmse

def calRMSE(src_pts, dst_pts, M, mask):
    # 求残差
    sum_H = 0  # 残差和
    num = 0  # 参与统计的总个数
    for i, j, m in zip(src_pts, dst_pts, mask):
        P_src = np.float32([i[0],i[1],1]).reshape((-1, 1))
        P = np.matmul(M, P_src)  # 通过计算出的矩阵预测点
        p = np.float32([P[0] / P[2], P[1] / P[2]])  # 从齐次矩阵变为2维点
        j = j.T
        # 范数
        distance = np.linalg.norm(p - j)
        if (m == True):
            sum_H += distance
            num += 1
    rmse = cmath.sqrt(sum_H / num)
    print("rmse : ", rmse)
    return rmse

def PLR(arrr):
    '''

    @param arrr:
    @return:
    '''
    temp = []
    for i in range(len(arrr)-1):
        if arrr[i+1]-arrr[i] >0:
            temp.append(1)
        elif  arrr[i+1]-arrr[i] ==0:
            temp.append(0)
        else:
            temp.append(-1)
    return temp

def PLR1(arrr):
    '''
    跟第一点的斜率
    @param arrr:
    @return:
    '''
    temp = []
    for i in range(len(arrr)-1):
        if arrr[i+1]-arrr[0] >0:
            temp.append(1)
        elif arrr[i+1]-arrr[i] ==0:
            temp.append(0)
        else:
            temp.append(-1)
    return temp

def compareline(arr1,arr2):
    '''
    
    @param arr1:
    @param arr2:
    @return:
    '''
    temp = 0
    for i in range(len(arr1)):
        if arr1[i] ==arr2[i]:
            temp+=1
    return temp/len(arr1)
