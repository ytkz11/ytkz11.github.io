def timestamp():
    '''
    生成时间戳
    :return:
    '''
    min_t = int(time.time())
    max_t = int(time.time() * 1000)
    return str(min_t), str(max_t)


def fabu_time(t):
    '''
    将时间戳转换成时间格式
    :param t:
    :return:
    '''
    timeArray = time.localtime(t)
    fabu_time = time.strftime("%Y-%m-%d %H:%M:%S", timeArray)
    return fabu_time


def save(dict, savepath):
    '''
    传入字典保存数据
    :param dict:
    :return:
    '''

    # 有些作品中没有标题导致我们再保存数据时会覆盖，所以我们对没有标题的作品进行命名
    num = str(random.randint(0, 1000))
    if dict['desc'] == '':
        name = dict['user_name'] + '的作品{}'.format(num) + '.mp4'
    else:
        name = dict['user_name'] + '的作品{}'.format(num) + '.mp4'

    video_url = dict['video_url']
    res = requests.get(video_url)
    file_name = os.path.join(savepath , dict['user_name'])
    if os.path.exists(file_name):
        with open(file_name + '/' + name, 'wb') as f:
           if res.status_code == 200 :
            f.write(res.content)
            print('{}下载完成！'.format(name))
    else:
        os.mkdir(file_name)
        with open(file_name + '/' + name, 'wb') as f:
            f.write(res.content)
            print('{}下载完成！'.format(name))
    return file_name

def find_sec_uid(user_url):

    response = requests.get(user_url)
    # 获取重定向后地址
    redirect_url = response.url
    # print(redirect_url)
    # 正则匹配规则
    reg = ".*sec_uid=(.*?)&"
    # 获取到sec_uid
    sec_uid = re.match(reg, redirect_url).group(1)
    return sec_uid

1.先把mp4转为ts文件。

2.再把ts文件重新合成为mp4文件

import subprocess
import os


# Get MP4 in the folder
#  先把，mp4转为ts文件。
#  再把ts文件重新合成为mp4文件

#  抖音窄屏
class Mp4mergetoMp4:
    def __init__(self, folder, n=1):
        self.folder = folder
        self.outpath = os.path.join(self.folder, 'out')
        if os.path.exists(self.outpath) == 0:
            os.mkdir(self.outpath)
        self.origin_mp4_to_ts()
        self.out = self.ts_to_mp4(n)

    def get_filetype_list(self, path, filetype):
        file = []
        for root, dirs, files in os.walk(path, topdown=True):
            for n in files:
                _, ending = os.path.splitext(n)
                if ending == filetype:
                    file.append(os.path.join(root, n))
        return file

    def origin_mp4_to_ts(self):
        mp4_list = self.get_filetype_list(self.folder, '.mp4')
        for i in range(len(mp4_list)):
            out = os.path.join(self.outpath, str(i) + '.ts')
            if os.path.exists(out) == 1:
                os.remove(out)
            subprocess.call(
                ['ffmpeg', '-i', mp4_list[i], '-c', 'copy', out])

    def ts_to_mp4(self, n):

        out = os.path.join(self.outpath, 'out.mp4')
        ziduan = 'concat:'
        ziduan1 = ''
        ts_list = self.get_filetype_list(self.outpath, '.ts')
        for i in range(len(ts_list)):
            ziduan1 += os.path.basename(ts_list[i]) + '|'
        ziduan1 = ziduan1 * n
        ziduan = ziduan + ziduan1[:-1]
        os.chdir(self.outpath)
        subprocess.call(['ffmpeg', '-i', ziduan, '-acodec', 'copy', '-vcodec', 'copy', '-absf', 'aac_adtstoasc', out])
        return out
